INSERT INTO adt.evaluaciones_movimiento 
(id_sesion, id_ejercicio, punto_evaluado, peso_manejado, reps_logradas, rpe_observado, tecnica_calificacion)
VALUES 
{{ 
  List1.listData.map((item, index) => {
    const fila = List1.currentItemsView[index];
    
    // Al estar dentro de {{ }}, NO uses {{ }} para los parámetros
    const idSesion = this.params.id_sesion; 
    const idEjerc = item.id;
    const punto = item.grupo;
    const peso = fila.inp_peso.text || 0;
    const reps = fila.inp_reps.text || 0;
    const rpe = fila.inp_rpe.text || 0;
    const tecnica = fila.sel_tecnica.selectedOptionValue || 0;

    return `(${idSesion}, ${idEjerc}, '${punto}', ${peso}, ${reps}, ${rpe}, ${tecnica})`;
  }).join(",") 
}};
